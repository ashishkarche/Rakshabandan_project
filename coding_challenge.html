// Function to generate subsets
void calcSubset(vector<int>& A, vector<vector<int>>& res,
                vector<int>& subset, int index) {
    res.push_back(subset);
    for (int i = index; i < A.size(); i++) {
        subset.push_back(A[i]);
        calcSubset(A, res, subset, i + 1);
        subset.pop_back();
    }
}

                                  List<Integer> subset,
